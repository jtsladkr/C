#include <stdlib.h>
#include <stdio.h>
#include <conio.h>
#include <time.h>

void disp_car(int car_number, int distance)
{
    int i; //변수 i 선언
    
    printf("CAR #%d:", car_number); //자동차 넘버 출력
    
    for (i = 0; i < distance / 10; i++) { i는 0이고 또한 i는 distance보다 작아야하며 i++은 증감연산자를 활용하여 계속 돌아감
        printf("*"); //계속 하나씩 *를  더해가며 출력해 자동차의 위치를 표시
    }
    
    printf("\n"); // 줄바꿈
}
