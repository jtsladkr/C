#include <stdlib.h>
#include <stdio.h>
#include <conio.h>
#include <time.h>

void disp_car(int car_number, int distance)
{
    int i;
    
    printf("CAR #%d:", car_number); // 자동차 번호 출력
    
    for (i = 0; i < distance / 10; i++) {
        printf("*"); // 거리에 비례하여 '*'를 출력하여 자동차의 위치를 표시
    }
    
    printf("\n"); // 줄바꿈
}
