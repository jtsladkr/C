#include <stdlib.h>
#include <stdio.h>

int main(void)
{
    system("dir"); // 시스템 명령어 'dir'을 실행하여 현재 디렉토리의 파일 및 폴더 목록을 출력합니다.
    
    printf("아무 키나 치세요\n"); // 사용자에게 아무 키나 입력하도록 안내합니다.
    
    _getch(); // 사용자로부터 키 입력을 받습니다. (Windows 환경에서만 사용 가능)
    
    system("cls"); // 시스템 명령어 'cls'를 실행하여 화면을 지웁니다. (콘솔 화면을 초기화)
    
    return 0;
}
