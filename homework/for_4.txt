//입력한 값만큼 구구단을 출력하는 프로그램

#include<stdio.h>
 
int main()
{
int a;
printf("몇단까지 출력할까요? : "); 
scanf("%d", &a); //몇단까지 출력하고싶은지
  
    for (int x = 1;  x == a; ++x) //a단까지 출력하게 설정
    {
        printf("구구단 %d 단\n", x); 
 
        for (int y = 1; y < 10; ++y)
        {
            printf("%d x %d : %d\n", x, y, x * y);
        }
 
        printf("\n");
    }
    
    return 0;
}
